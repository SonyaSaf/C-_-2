#include <iostream>
#include <vector>

using namespace std;

int resss(const vector<int>& arr) {
    int result = 1;

    for (int i = 0; i < arr.size() && arr[i] <= result; ++i) {
        result += arr[i];
    }

    return result;
}

int main() {
    int n;
    cout << "Введите количество элементов в массиве: ";
    cin >> n;

    vector<int> array;
    cout << "Введите элементы массива через пробел: ";
    for (int i = 0; i < n; ++i) {
        int element;
        if (cin >> element) {
            array.push_back(element);
        }
    }
    int result = resss(array);

    cout << "Наименьшее положительное недостижимое число: " << result << endl;

    return 0;
}
