#include <iostream>
#include <vector>

using namespace std;

int main() {
    const int maxSize = 100001;
    vector<int> mas(maxSize, 0);

    int N, a, b, c, res = 0;
    cin >> N;

    for (int i = 0; i <= N; ++i) {
        cin >> a >> b >> c;
        mas[a] += c;
        mas[b + 1] -= c;
    }

    for (int i = 1; i <= N; ++i) {
        res += mas[i];
        cout << res << ' ';
    }

    cout << endl;

    return 0;
}
