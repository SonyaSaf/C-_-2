#include <iostream>

using namespace std;

bool proverka(const int arr1[], int size1, const int arr2[], int size2) {
    int i = 0;
    int j = 0; 

    while (i < size1 && j < size2) {
        if (arr1[i] == arr2[j]) {
            i++;
        }
        j++;
    }

    return i == size1;
}

int main() {
    cout << "Введите размер первого массива: ";
    int size1;
    cin >> size1;

    cout << "Введите элементы первого массива через пробел: ";
    int* arr1 = new int[size1];
    for (int i = 0; i < size1; ++i) {
        cin >> arr1[i];
    }

    cout << "Введите размер второго массива: ";
    int size2;
    cin >> size2;

    cout << "Введите элементы второго массива через пробел: ";
    int* arr2 = new int[size2];
    for (int i = 0; i < size2; ++i) {
        cin >> arr2[i];
    }

    if (proverka(arr1, size1, arr2, size2)) {
        cout << "Yes" << endl;
    }
    else {
        cout << "No" << endl;
    }

    return 0;
}
