#include <iostream>

using namespace std;

int gcd(int a, int b) {
    while (b != 0) {
        int i = b;
        b = a % b;
        a = i;
    }
    return a;
}

int main() {
    int N, M;
    cin >> N >> M;

    int D = gcd(N - 1, M - 1);

    int res = (N - 1) + (M - 1) - D;

    cout << res << endl;

    return 0;
}
