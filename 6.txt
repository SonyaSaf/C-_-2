#include <iostream>
#include <cmath>

using namespace std;

int main() {
    double R;
    cin >> R;

    int s = 1000;

    int p = 0;
    for (int i = 0; i < s; ++i) {
        for (int j = 0; j < s; ++j) {
            double x = i + 0.5;
            double y = j + 0.5;

            double d = sqrt(pow(x - s / 2.0, 2) + pow(y - s / 2.0, 2));

            if (d <= R + 0.5) {
                p++;
            }
        }
    }

    cout << p << endl;

    return 0;
}
