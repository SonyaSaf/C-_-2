#include <iostream>
#include <vector>

using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> g(N);
    for (int i = 0; i < N; ++i) {
        cin >> g[i];
    }
    vector<int> p(N, 0);
    p[0] = max(0, g[0]);
    for (int i = 1; i < N; ++i) {
        p[i] = max(p[i-1] + g[i], g[i]);
    }
    int res = p[0];
    for (int i = 1; i < N; ++i) {
        res = max(res, p[i]);
    }

    cout << res << endl;

    return 0;
}
